# Sample proxy configuration
# Format 1: Basic
127.0.0.1:1080

# Format 2: ABC proxy format (host:port:user:pass)
da579d826694e90c.abcproxy.vip:4950:abc5990713_oy38-zone-abc:54146081

# Format 3: user:pass@host:port
username:password@proxy.example.com:1080

# Format 4: With protocol prefix
socks5://127.0.0.1:1080
socks5://username:password@proxy.example.com:1080
http://proxy.example.com:8080
socks4://127.0.0.1:1080
